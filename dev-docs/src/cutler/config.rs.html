7/// Returns the path to the configuration file by checking several candidate locations.
8pub fn get_config_path() -> PathBuf {
9    let mut candidates = Vec::new();
10
11    // Decide candidates in order.
12    if let Some(xdg_config) = env::var_os("XDG_CONFIG_HOME") {
13        let candidate = PathBuf::from(xdg_config).join("cutler").join("config.toml");
14        candidates.push(candidate);
15    }
16
17    if let Some(home) = env::var_os("HOME") {
18        let candidate = PathBuf::from(&home)
19            .join(".config")
20            .join("cutler")
21            .join("config.toml");
22        candidates.push(candidate);
23
24        let candidate2 = PathBuf::from(home).join(".config").join("cutler.toml");
25        candidates.push(candidate2);
26    }
27
28    candidates.push(PathBuf::from("config.toml"));
29
30    // Return the first candidate that exists.
31    // If none exists, return the first candidate (this may lead to a prompt to create an example config).
32    for candidate in &candidates {
33        if candidate.exists() {
34            return candidate.to_owned();
35        }
36    }
37    candidates
38        .first()
39        .cloned()
40        .unwrap_or_else(|| PathBuf::from("config.toml"))
41}
