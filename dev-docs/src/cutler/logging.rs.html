1/// Color constants.
2pub const RED: &str = "\x1b[31m";
3pub const GREEN: &str = "\x1b[32m";
4pub const YELLOW: &str = "\x1b[33m";
5pub const PINK: &str = "\x1b[35m";
6pub const RESET: &str = "\x1b[0m";
7pub const BOLD: &str = "\x1b[1m";
8
9/// Log levels for printing messages.
10#[derive(PartialEq)]
11pub enum LogLevel {
12    Success,
13    Error,
14    Warning,
15    Info,
16    CommandOutput,
17}
18
19/// Central logging function.
20pub fn print_log(level: LogLevel, message: &str) {
21    match level {
22        LogLevel::Success => println!("{}[SUCCESS]{} {}", GREEN, RESET, message),
23        LogLevel::Error => eprintln!("{}[ERROR]{} {}", RED, RESET, message),
24        LogLevel::Warning => eprintln!("{}[WARN]{} {}", YELLOW, RESET, message),
25        LogLevel::Info => println!("{}[INFO]{} {}", BOLD, RESET, message),
26        LogLevel::CommandOutput => println!("{}[CMD OUTPUT]{} {}", PINK, RESET, message),
27    }
28}
