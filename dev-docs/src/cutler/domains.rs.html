15/// Recursively flattens a TOML table into a list of (domain, settings) pairs.
16/// For example, [menuextra.clock] becomes ("menuextra.clock", { … }).
17pub fn flatten_domains(
18    prefix: Option<String>,
19    table: &toml::value::Table,
20    dest: &mut Vec<(String, toml::value::Table)>,
21) {
22    let mut flat_table = toml::value::Table::new();
23
24    // Process all non-table values in one pass
25    for (key, value) in table {
26        if let Value::Table(inner) = value {
27            // Create new prefix for nested table
28            let new_prefix = match &prefix {
29                Some(p) if !p.is_empty() => format!("{}.{}", p, key),
30                _ => key.clone(),
31            };
32
33            // Process nested table recursively
34            flatten_domains(Some(new_prefix), inner, dest);
35        } else {
36            // Add non-table values to flat table
37            flat_table.insert(key.clone(), value.clone());
38        }
39    }
40
41    // Only add if there are non-table values
42    if !flat_table.is_empty() {
43        dest.push((prefix.unwrap_or_default(), flat_table));
44    }
45}

47/// Given the flattened domain (from config) and a key, return the effective domain and key.
48///
49/// • If the domain is not beginning with "NSGlobalDomain", returns ("com.apple.{domain}", key)
50/// • For an entry exactly equal to "NSGlobalDomain", returns ("NSGlobalDomain", key)
51/// • For an entry starting with "NSGlobalDomain.", returns ("NSGlobalDomain", "{rest-of-domain}.{key}")
52pub fn get_effective_domain_and_key(domain: &str, key: &str) -> (String, String) {
53    if domain == "NSGlobalDomain" {
54        ("NSGlobalDomain".to_string(), key.to_string())
55    } else if let Some(remainder) = domain.strip_prefix("NSGlobalDomain.") {
56        if remainder.is_empty() {
57            ("NSGlobalDomain".to_string(), key.to_string())
58        } else {
59            (
60                "NSGlobalDomain".to_string(),
61                format!("{}.{}", remainder, key),
62            )
63        }
64    } else {
65        (format!("com.apple.{}", domain), key.to_string())
66    }
67}

78/// Collects domains and their flattened settings from a parsed TOML configuration.
79/// Returns a HashMap where keys are domain strings and values are settings tables.
80pub fn collect_domains(
81    parsed: &toml::Value,
82) -> Result<HashMap<String, toml::value::Table>, Box<dyn std::error::Error>> {
83    let root_table = parsed
84        .as_table()
85        .ok_or("Invalid config format: expected table at top level")?;
86
87    let mut domains = HashMap::new();
88
89    for (key, value) in root_table {
90        if key == "external" {
91            continue;
92        }
93
94        if let Value::Table(inner) = value {
95            // Pre-allocate with reasonable capacity to avoid resizes
96            let mut flat = Vec::with_capacity(inner.len() * 2);
97            flatten_domains(Some(key.clone()), inner, &mut flat);
98
99            // Move items directly into the HashMap
100            for (domain, table) in flat {
101                domains.insert(domain, table);
102            }
103        }
104    }
105
106    Ok(domains)
107}

109/// Helper function to check if a domain exists
110fn domain_exists(full_domain: &str) -> bool {
111    let cache = DOMAIN_CACHE.lock().unwrap();
112    if let Some(domains) = &*cache {
113        domains.contains(full_domain)
114    } else {
115        // Fallback to direct check
116        Command::new("defaults")
117            .arg("read")
118            .arg(full_domain)
119            .output()
120            .map(|o| o.status.success())
121            .unwrap_or(false)
122    }
123}

125/// Checks whether a given domain exists using the "defaults" command.
126pub fn check_domain_exists(full_domain: &str) -> Result<(), Box<dyn std::error::Error>> {
127    // Attempt to initialize cache if needed
128    INIT.call_once(|| {
129        let output = match Command::new("defaults").arg("domains").output() {
130            Ok(output) => output,
131            Err(e) => {
132                print_log(
133                    LogLevel::Warning,
134                    &format!(
135                        "Failed to fetch domains: {}, some domain checks may fail",
136                        e
137                    ),
138                );
139                return; // Cache remains null, indicating that the user should be careful with domain checks
140            }
141        };
142
143        if output.status.success() {
144            let domains_str = String::from_utf8_lossy(&output.stdout);
145            let domains: HashSet<String> = domains_str
146                .split(|c: char| c == ',' || c.is_whitespace())
147                .map(|s| s.trim().to_owned())
148                .filter(|s| !s.is_empty())
149                .collect();
150
151            let mut cache = DOMAIN_CACHE.lock().unwrap();
152            *cache = Some(domains);
153        }
154    });
155
156    if domain_exists(full_domain) {
157        Ok(())
158    } else {
159        Err(format!("Domain '{}' does not exist. Aborting.", full_domain).into())
160    }
161}

168/// Helper: Reads the current value from defaults (if any) for a given effective domain and key.
169pub fn get_current_value(eff_domain: &str, eff_key: &str) -> Option<String> {
170    let output = Command::new("defaults")
171        .arg("read")
172        .arg(eff_domain)
173        .arg(eff_key)
174        .output()
175        .ok()?;
176    if !output.status.success() {
177        return None;
178    }
179    let s = String::from_utf8_lossy(&output.stdout).trim().to_string();
180    if s.is_empty() { None } else { Some(s) }
